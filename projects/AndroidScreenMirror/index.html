📦 SDK 模块结构

screenshare-sdk/src/main/java/com/screenshare/sdk/
├── ScreenshareSDK.java         # SDK 唯一入口类
├── Sender.java                # 发送端（状态机）
├── Receiver.java              # 接收端（状态机）
├── SenderConfig.java          # 发送端配置
├── ReceiverConfig.java        # 接收端配置
├── Common/
│   ├── AtomicBuffer.java      # 原子化预分配缓冲区（GC优化）
│   ├── ErrorCode.java         # 错误码体系（1xxx-5xxx）
│   ├── SenderState.java       # 发送端状态机
│   ├── ReceiverState.java     # 接收端状态机
│   ├── BandwidthAdapter.java  # 带宽自适应
│   ├── ConnectionStateMachine.java
│   ├── HeartbeatMonitor.java  # 心跳监控
│   ├── LatencyTester.java     # 延迟测试
│   └── ReconnectionManager.java # 重连管理
├── capture/
│   └── ScreenCapturer.java    # 屏幕采集（MediaProjection）
├── codec/
│   ├── VideoEncoder.java      # 硬件 H.264/H.265 编码
│   ├── VideoDecoder.java      # 硬件 H.264/H.265 解码
│   ├── VideoDecoderIntegrator.java
│   ├── AudioCapture.java      # 音频采集
│   ├── AudioEncoder.java      # 音频编码
│   ├── AudioDecoder.java      # 音频解码
│   └── AudioPlayer.java       # 音频播放
├── network/
│   ├── UdpChannel.java        # UDP 通道
│   └── RtpSession.java        # RTP 打包/解包（FU-A 分片）
├── service/
│   ├── ScreenCaptureService.java  # 前台服务
│   └── ScreenMirroringService.java
├── touch/
│   ├── TouchEncoder.java      # 触摸编码（独立通道）
│   ├── TouchDecoder.java      # 触摸解码
│   └── TouchInjectorService.java  # AccessibilityService 注入
└── wifi/
    ├── P2pConnectionManager.java   # WiFi P2P 管理
    ├── P2pBroadcastReceiver.java  # P2P 广播接收
    └── ServiceDiscovery.java       # 服务发现协议
